Script TryIssueReport {
cmd "sym set _UserReportType_ $_MArg1_"
cmd "if sym ReportCardCount == 0 menucmd IssueReportNoReportCards"
cmd "if sym ReportCardCount > 0 menucmd IssueReportConfirm"
}
Script IssueReportNoReportCards {
cmd "sym set DialogBoxTitle DialogTitleNoReportCards"
cmd "sym set DialogBoxMessage DialogMessageNoReportCards"
cmd "sym set DialogMenuCmdOnButton1 MP_NOP"
cmd "loadmenu DialogBoxOk.txt /fadein 0.15"
}
Script IssueReportConfirm {
cmd "sym set DialogBoxTitle DialogTitleUserReportConf"
cmd "sym set DialogBoxMessage DialogMessageUserReportConf"
cmd "sym set DialogMenuCmdOnButton1 IssueReportConfirm_Go"
cmd "sym set DialogMenuCmdOnButton0 MP_NOP"
cmd "sym set DialogBoxButton2Used 0"
cmd "loadmenu DialogBox.txt"
}
Script IssueReportConfirm_Go {
ignoreInputTime 0.1
cmd "clearmenu DialogReport"
cmd "wallet reportUser _FriendName_ $_UserReportType_"
cmd "timer setAsyncWait 1"
}
