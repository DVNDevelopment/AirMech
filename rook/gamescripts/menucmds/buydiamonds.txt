}
Script SelectPaymentBackProvider {
ignoreInputTime 0.2
cmd "gameui disableAll BuyDiamondsPaymentOptions"
cmd "gameui hideGroup PArrow0 PArrow1 PArrow2"
cmd "addUIAnim BuyDiamondsCharacters mas_D2PackReset"
cmd "gameui disableGroup BuyDiamondsBackButton BuyDiamondsNote BuyDiamondsCharacters"
cmd "if sym BuyDiamondsSelection == D2Pack5 menucmd SelectPaymentBackD2Pack5"
cmd "if sym BuyDiamondsSelection == D2Pack10 menucmd SelectPaymentBackD2Pack10"
cmd "if sym BuyDiamondsSelection == D2Pack20 menucmd SelectPaymentBackD2Pack20"
cmd "if sym BuyDiamondsSelection == D2Pack50 menucmd SelectPaymentBackD2Pack50"
cmd "if sym BuyDiamondsSelection == D2Pack100 menucmd SelectPaymentBackD2Pack100"
cmd "if sym BuyDiamondsSelection == D2Pack200 menucmd SelectPaymentBackD2Pack200"
cmd "sym set BuyDiamondsSelection none"
}
