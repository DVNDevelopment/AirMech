Script ArchivesTimelineMove {
ignoreInputTime 0.3
cmd "gameui getLastRenderPos #ArchivesTimeline@_ArchivesCurrent@Orb@_MArg1_@ _TimelineMoveToX_ _TimelineMoveToY_"
cmd "sym dec _TimelineMoveToX_ 6"
cmd "sym set ArchivesTimelineHintHidden 1"
cmd "pcmd_seq Typing 0 addUIAnim ArchivesTimelineCurrent mas_ArchivesTimelineCurrentMove"
cmd "pcmd_seq Typing 0.1 sym setFrom _TimelineMoveFromX_ _TimelineMoveToX_"
cmd "pcmd_seq Typing 0.25 gameui enableAll ArchivesTimelineCurrent"
cmd "if sym $_MArg2_ == 1 pcmd 0.1 addUIAnim ArchivesTimelineHint mas_ArchivesTimelineHintShow"
}
Script ArchivesNext {
ignoreInputTime 0.3
rcmd "console flushAllPendingSeq Typing"
cmd "gameui disableGroup  ArchivesTopSectionType ArchivesTopYearRangeType ArchivesTopCount1 ArchivesTopCount2 ArchivesTopCount3"
cmd "pcmd_seq Typing 0.1 floatsym set S_ArchEra 0.0"
cmd "pcmd_seq Typing 1.6 floatsym set S_ArchYearRange 0.0"
cmd "if sym _ArchivesCurrent == Decay sym set _ArchivesCurrent Over"
cmd "if sym _ArchivesCurrent == Decay pcmd_seq Typing 0.2 sym set _ArchivesEra era_name_overgrowth"
cmd "if sym _ArchivesCurrent == Decay pcmd_seq Typing 0.2 sym set _ArchivesYearRange Archives_OverYearRange"
cmd "if sym _ArchivesCurrent == Decay pcmd 0.2 gameui enable ArchivesTopCount2"
cmd "if sym _ArchivesCurrent == Over sym set _ArchivesCurrent States"
cmd "if sym _ArchivesCurrent == Over pcmd_seq Typing 0.2 sym set _ArchivesEra era_name_thousandstates"
cmd "if sym _ArchivesCurrent == Over pcmd_seq Typing 0.2 sym set _ArchivesYearRange Archives_StatesYearRange"
cmd "if sym _ArchivesCurrent == Over pcmd 0.2 gameui enable ArchivesTopCount3"
cmd "addUIAnim ArchivesTopSection mas_ArchivesTimelineSectionSwitch"
cmd "addUIAnim ArchivesTopYearRange mas_ArchivesTimelineYearRangeSwitch"
cmd "gameui disableAll ArchivesTimelineCurrent"
cmd "addUIAnim ArchivesTimelineHint mas_ArchivesTimelineHintHide"
cmd "gameui disableGroup ArchivesDecayBg ArchivesOverBg ArchivesStatesBg"
cmd "if sym _ArchivesCurrent == Decay gameui enable ArchivesOverBg"
cmd "if sym _ArchivesCurrent == Over gameui enable ArchivesStatesBg"
cmd "addUIAnim ArchivesDecayTimeline mas_ArchivesTimelineShiftRight"
cmd "addUIAnim ArchivesOverTimeline mas_ArchivesTimelineShiftRight"
cmd "addUIAnim ArchivesStatesTimeline mas_ArchivesTimelineShiftRight"
cmd "if sym _ArchivesCurrent == Decay pcmd_seq Typing 0.3 gameui directPushName ArchivesTimelineOverOrb0"
cmd "if sym _ArchivesCurrent == Over pcmd_seq Typing 0.3 gameui directPushName ArchivesTimelineStatesOrb0"
cmd "pcmd_seq Typing 0.3 gameui enable ArchivesTopSectionType"
cmd "pcmd_seq Typing 1.6 gameui enable ArchivesTopYearRangeType"
}
