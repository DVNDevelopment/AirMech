Script GetReadyArmySlot0 {
cmd "log MenuCmd: GetReadyArmySlot0"
#if vr
cmd "sym set FinalUnitToBuild $AS0P@C_BINDING@"
cmd "sym sprintf FinalUnitToBuildST DispUnit%s $AS0P@C_BINDING@"
#else
cmd "sym set FinalUnitToBuild $ArmySlot0Custom@C_BINDING@"
cmd "sym sprintf FinalUnitToBuildST DispUnit%s $ArmySlot0Custom@C_BINDING@"
#endif
cmd "gameUI disableAll #StatContainer@ArmySlotLastSelection@"
cmd "sym set ArmySlotLastSelection ArmySlot0"
cmd "gameUI enableAll StatContainerArmySlot0"
cmd "if sym QuickOrderTab == 1 gameUI disable #@OrderSlotLastSelection@Quick #@ArmySlotLastSelection@Quick"
cmd "if sym QuickOrderTab == 1 gameUI enable ArmySlot0Quick"
}
