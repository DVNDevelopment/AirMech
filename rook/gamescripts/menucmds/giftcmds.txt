cmd "pcmd 0.9 gameUI disable CapsuleRareMiddle"
cmd "pcmd 0.9 gameUI disable CapsuleRareBottom"

cmd "pcmd 1.2 addUIAnim ChoiceButton2 mas_Choice2MoveIn"

cmd "pcmd 1.2 addUIAnim ChoiceTitle2 mas_ChoiceTitle"

cmd "pcmd t3 if literal $_MArg1_ == amlvlup menucmd CommonAMLvlUpDialogLoad local $_MArg2_ $_MArg3_"
cmd "pcmd t1 if literal $_MArg1_ == reqcnt menucmd CommonAMReqCntDialogLoad"
cmd "pcmd t1 if literal $_MArg1_ == victory menucmd CommonVictoryDialogLoad"
cmd "pcmd t6 if literal $_MArg1_ == leaderboard menucmd CommonLeaderboardDialogLoad local $_MArg2_ $_MArg3_ $_MArg4_ $_MArg5_ $_MArg6_"
#else
rcmd "if literal $_MArg1_ == questReward runnow sym set _EndGameSeqMsg_ 1"
cmd "if literal $_MArg1_ == questReward menucmd CommonQuestRewardDialogLoad local $_MArg2_ $_MArg3_"
#endif
}
