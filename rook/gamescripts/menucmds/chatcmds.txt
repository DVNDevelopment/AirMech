cmd "if sym GL_ChatMinimized == 0 if !ingame sym set _ChatPosX 0"

cmd "if sym GL_ChatMinimized == 2 sym set _ChatPosX 570"
cmd "if sym GL_ChatMinimized == 2 adduianim ChatShowButton mas_ChatTabShow"
cmd "if sym GL_ChatMinimized == 0 adduianim ChatShowButton mas_ChatTabHide"
cmd "if sym GL_ChatMinimized == 0 gameui hide ChatShowButton"
cmd "sym set _ChatPosY 0"
cmd "if sym GL_ChatMinimized == 2 pcmd 0.2 gameui show ChatShowButton"
cmd "if sym GL_ChatMinimized == 2 pcmd 0.2 gameUI enable ChatShowButton"
cmd "if sym GL_ChatMinimized != 2 pcmd 0.2 gameUI disable ChatShowButton"
cmd "if ingame pcmd t2 addUIAnim InGameGlobalChat mas_ChatPosition"
cmd "if !ingame pcmd t2 addUIAnim MessageChatContainer mas_ChatPosition"
}
