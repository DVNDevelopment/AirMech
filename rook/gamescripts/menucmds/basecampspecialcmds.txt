cmd "pcmd 0 menucmd InitSpSymbols"
cmd "pcmd 0 sym set map_cfg $_MArg3_"

}
Script BSC_RunMap_Coop {

}
Script BSC_RunMap {

#if vr
rcmd "sym set _EndGameSeqMsg_ 1"
#endif
}
