StateController Lvl_Duel {
#include Lvl_CommonStates
State initial {
In 0.0 "sym set MatchOver 0"
In 0.0 "sym setNSet 60 UpkeepTeam0_ -1"
In 0.0 "sym setNSet 60 UpkeepTeam1_ -1"
In 0.0 "sym set _UTInc_ 0"
}
State Init {
}
State EndPreGame {
DoL 240.0 |rook doLevelState SpawnGroupA
DoL 120.0 |rook doLevelState SpawnGroupB
}
State Tick {
Cfg timeBlock 0.25
Do 0.0 "if sym MatchOver == 0 rook doLevelState NormalSpawn"
DoL 0.0 if sym MatchOver == 0 rook doLevelState UpkeepTrack
}
State NormalSpawn {
DoL 0.0 if gameValidation spCr_p0 rook doLevelState CreepSpawnTeam0
DoL 0.0 if gameValidation spCr_p1 rook doLevelState CreepSpawnTeam1
}
State SpawnGroupA {
DoL ~0.0 |state .sideA2 do ProduceNeutralUnit MissileTank sideA2
DoL ~0.0 |state .sideA3 do ProduceNeutralUnit MissileTank sideA3
DoL ~0.0 |state .sideA4 do ProduceNeutralUnit Tank sideA4
DoL ~0.0 |state .sideB2 do ProduceNeutralUnit MissileTank sideB2
DoL ~0.0 |state .sideB3 do ProduceNeutralUnit MissileTank sideB3
DoL ~0.0 |state .sideB4 do ProduceNeutralUnit Tank sideB4
}
State SpawnGroupB {
DoL ~0.0 |state .sideA1 do ProduceNeutralUnit BankSmallSp1 sideA1
DoL ~0.0 |state .sideB1 do ProduceNeutralUnit BankSmallSp1 sideB1
DoL 120.0 |rook doLevelState SpawnGroupB
}
State CreepSpawnTeam0 {
DoL 0.0 |rook spawnCreepBegin 0
DoL 1.0 |rook spawnCreep 0	/type 0
DoL 1.5 |rook spawnCreep 0 	/type 0
DoL 2.0 |rook spawnCreep 0 	/type 0
DoL 2.5 |rook spawnCreep 0 	/type 1
DoL 5.0 |rook spawnCreepEnd 0
}
State CreepSpawnTeam1 {
DoL 0.0 |rook spawnCreepBegin 1
DoL 1.0 |rook spawnCreep 1 	/type 0
DoL 1.5 |rook spawnCreep 1	/type 0
DoL 2.0 |rook spawnCreep 1	/type 0
DoL 2.5 |rook spawnCreep 1	/type 1
DoL 5.0 |rook spawnCreepEnd 1
}
}
