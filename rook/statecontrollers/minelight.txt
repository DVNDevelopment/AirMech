StateController MineLight {

}
State MineArmStart {

Do 0.0 "anim play 0 units/MineLight/Idle.aaf stopandhold 0.2"
}

Do 0.0 "pfx boneLatchNRTSpawnIfVis MineArm> bone_middle"
}

Do 0.0 "pfx boneLatchNRTSpawnIfVis MineBlink> bone_middle"
}

}

}

}
